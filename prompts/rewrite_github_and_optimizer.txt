--- fetch_github_digest() replacement ---

Signature: async def fetch_github_digest(url_or_username: str) -> GitHubDigest | None

Steps:
1. Extract username from url_or_username using existing _extract_username()
2. Fetch pinned repos via GraphQL:
     POST https://api.github.com/graphql
     query: { user(login: $username) {
               pinnedItems(first: 6, types: REPOSITORY) {
                 nodes { ... on Repository { name } }
               }
             }}
     Use Bearer token from _headers() if available.
     On any failure (no token, GraphQL error, network error): set pinned_names = []
     Never let this block the rest of the fetch.
3. Fetch all repos:
     GET /users/{username}/repos?per_page=100&sort=updated
     Follow Link header pagination to get ALL pages — do not stop at 100.
     Filter: keep only repos where fork == False AND visibility == "public"
             (or where private == False if visibility key absent)
     total_repos = count of all repos returned before fork filter
4. For each non-fork public repo call _fetch_repo_digest_full() (defined below)
     Use asyncio.gather() across all repos concurrently.
     On exception for a single repo: log warning, skip that repo, continue.
5. Build top_languages across all repos (same logic as existing code)
6. Return GitHubDigest(username, repos, top_languages, pinned_repos=pinned_names,
                       total_repos=total_repos)

--- GitHubDigest.to_context_string() replacement ---

This method serializes everything for Claude. Use a per-repo token budget to ensure
the total string fits in Claude's extended context without cutting off mid-repo.

Token budget logic:
  TOTAL_CHAR_BUDGET = 180_000   # ~45k tokens, safe for Claude's 200k context window
                                 # leaving room for resume + JD + system prompt
  repos_count = len(self.repos)
  if repos_count == 0: return "No GitHub data."

  # Give pinned repos 2x budget weight
  # Allocate proportionally, minimum 1000 chars per repo
  pinned = set(self.pinned_repos)
  base_budget = TOTAL_CHAR_BUDGET // repos_count
  budgets = {}
  for r in self.repos:
      budgets[r.name] = min(base_budget * 2, 12000) if r.name in pinned \
                        else min(base_budget, 8000)
  # Scale down if total exceeds budget
  total_allocated = sum(budgets.values())
  if total_allocated > TOTAL_CHAR_BUDGET:
      scale = TOTAL_CHAR_BUDGET / total_allocated
      budgets = {k: max(1000, int(v * scale)) for k, v in budgets.items()}

Serialization format per repo (write a helper _serialize_repo(r, budget) -> str):

async def optimize_with_digest(
    resume_text: str,
    job_description: str,
    github_digest: "GitHubDigest | None" = None,
) -> ATSOptimizeResponse:
    """
    Full-depth ATS optimizer. Claude receives the complete resume, complete JD,
    and complete GitHub context serialized from GitHubDigest.to_context_string().
    No pre-filtering. No Python-side scoring. One Claude call.
    """
    client = anthropic.AsyncAnthropic(api_key=settings.anthropic_api_key)

    github_context = (
        github_digest.to_context_string()
        if github_digest is not None
        else "No GitHub data provided."
    )

    user_msg = _DIGEST_USER.format(
        resume_text=resume_text,
        job_description=job_description,
        github_context=github_context,
    )

    try:
        message = await client.messages.create(
            model=settings.anthropic_model,
            max_tokens=6000,
            system=_DIGEST_SYSTEM,
            messages=[{"role": "user", "content": user_msg}],
        )
    except anthropic.APIError as exc:
        logger.error("Anthropic API error (digest optimize): %s", exc)
        raise

    full = message.content[0].text if message.content else ""

    # --- Parse structured XML output ---

    optimized_text = _extract_tag(full, "optimized_resume") or resume_text

    def _safe_int(tag: str, fallback: int) -> int:
        raw = _extract_tag(full, tag).strip()
        try:
            return max(0, min(100, int(raw)))
        except (ValueError, TypeError):
            return fallback

    score_before = _safe_int("ats_score_before", 0)
    score_after  = _safe_int("ats_score_after",  0)

    def _csv_tag(tag: str) -> list[str]:
        raw = _extract_tag(full, tag)
        if not raw:
            return []
        return [k.strip() for k in raw.split(",") if k.strip()][:40]

    matched  = _csv_tag("matched_keywords")
    missing  = _csv_tag("missing_keywords")

    transparency  = _extract_tag(full, "transparency_report")
    gap           = _extract_tag(full, "gap_analysis")
    interview     = _extract_tag(full, "interview_prep")
    role_reframes = _parse_role_reframes(_extract_tag(full, "role_reframing"))

    if role_reframes:
        transparency += "\n\nRole Titles Reframed:\n" + "\n".join(
            f"- {r['original']} → {r['reframed']}: {r['justification']}"
            for r in role_reframes
        )

    changes_raw = ""
    if "---CHANGES---" in full:
        changes_raw = full.split("---CHANGES---", 1)[1].strip()
    change_items, changes_plain = _parse_changes(changes_raw)

    latex_text = await _generate_latex(optimized_text, client)

    return ATSOptimizeResponse(
        original_text=resume_text,
        optimized_text=optimized_text,
        latex_text=latex_text,
        changes_summary=changes_plain,
        change_items=change_items,
        ats_score_before=score_before,
        ats_score_after=score_after,
        matched_keywords=matched,
        missing_keywords=missing,
        improvements=[],
        transparency_report=transparency,
        interview_prep=interview,
        gap_analysis=gap,
        linkedin_unavailable=False,
        role_reframes=role_reframes,
    )
